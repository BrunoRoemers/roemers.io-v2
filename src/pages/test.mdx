```jsx
import { MDXProvider } from '@mdx-js/react'
import * as React from 'react'
import CodeBlock from './code-block'
import Container from './container'
import Theme from './theme'

interface Props {
  children?: React.ReactNode,
}

// example see https://github.com/gatsbyjs/gatsby-starter-mdx-basic/blob/master/src/components/layout.js
const MarkdownPage = ({ children }: Props) => {
  return (
    <Theme>
      <Container>
        <article>
          <MDXProvider components={{
            // docs, see: https://www.gatsbyjs.com/docs/how-to/routing/customizing-components/
            h1: props => <h1 {...props} className='text-red-600'/>,
            code: ({children, className}) => <CodeBlock language={className.replace('language-', '')}>{children.trim()}</CodeBlock>,
          }}>
            {children}
          </MDXProvider>
        </article>
      </Container>
    </Theme>
  )
}

export default MarkdownPage
```
